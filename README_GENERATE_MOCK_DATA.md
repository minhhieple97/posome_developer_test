## Problem 2
```js
function arr_rand(a) {
	return a[Math.floor(Math.random() * a.length)];
}

function int_rand(min, max) {
	return min + Math.floor(Math.random() * max)
}

function int_rand_exclude(min, max, exclude) {
	const rnd = int_rand(min, max)
	return rnd != exclude ? rnd : int_rand_exclude(min, max, exclude)
}

function arr_rand_exclude(a, excludes) {
	let item = arr_rand(a)
    return excludes[item] ? arr_rand_exclude(a, excludes) : item
}
```
